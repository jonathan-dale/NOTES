  ### deploy a microservice app to cluster
git clone https://github.com/linuxacademy/robot-shop.git
k create namespace robot-shop
k -n robot-shop create -f ~/k8s/robot-shop/k8s/descriptors/
k get pods -n robot-shop -w

 ### After the pods come up you can browse to the app on the public IP of the cluster nodes port 30080
k get nodes -o wide
http://$kube_server_public_ip:30080
